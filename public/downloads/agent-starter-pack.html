// _shared/claude.ts — Claude API wrapper
export async function askClaude(
  systemPrompt: string,
  userMessage: string,
  options?: { maxTokens?: number; temperature?: number }
): Promise<string> {
  const response = await fetch("https://api.anthropic.com/v1/messages", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      "x-api-key": Deno.env.get("ANTHROPIC_API_KEY")!,
      "anthropic-version": "2023-06-01",
    },
    body: JSON.stringify({
      model: "claude-sonnet-4-20250514",
      max_tokens: options?.maxTokens || 4096,
      temperature: options?.temperature || 0.7,
      system: systemPrompt,
      messages: [{ role: "user", content: userMessage }],
    }),
  });

  const data = await response.json();
  return data.content[0].text;
}

// _shared/resend.ts — Email sending
export async function sendEmail(
  to: string,
  subject: string,
  html: string,
  from?: string
): Promise<void> {
  await fetch("https://api.resend.com/emails", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      "Authorization": `Bearer ${Deno.env.get("RESEND_API_KEY")}`,
    },
    body: JSON.stringify({
      from: from || "you@yourdomain.com",
      to,
      subject,
      html,
    }),
  });
}

Agent 1: Morning Briefer

Your personal AI chief of staff — delivers a comprehensive daily briefing to your inbox every morning before you wake up.

Trigger: Cron (daily 7am) Output: Email Difficulty: Beginner Value: 30 min/day saved

What It Does

Every morning at 7am, this agent queries your database for yesterday's key metrics (revenue, new subscribers, content performance), checks your calendar context, and generates a concise briefing email. You wake up to a summary of everything that happened while you slept, plus recommended priorities for the day.

Database Schema

CREATE TABLE daily_briefings (
  id            uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  briefing_date date NOT NULL UNIQUE,
  metrics       jsonb NOT NULL,      -- {revenue, subscribers, pageviews, etc.}
  highlights    jsonb NOT NULL,      -- notable events from yesterday
  priorities    jsonb NOT NULL,      -- AI-recommended focus areas
  full_briefing text NOT NULL,       -- the rendered email content
  sent_at       timestamptz,
  created_at    timestamptz DEFAULT now()
);

-- Also queries from these existing tables:
-- revenue_events (amount, source, created_at)
-- email_subscribers (created_at for new signups)
-- content_pipeline (status, published_at, engagement)

Edge Function Code

// supabase/functions/morning-briefer/index.ts
import { serve } from "https://deno.land/std@0.168.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";
import { askClaude } from "../_shared/claude.ts";
import { sendEmail } from "../_shared/resend.ts";
import { readBrain } from "../_shared/brain.ts";

serve(async (req) => {
  const supabase = createClient(
    Deno.env.get("SUPABASE_URL")!,
    Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!
  );

  const yesterday = new Date();
  yesterday.setDate(yesterday.getDate() - 1);
  const dateStr = yesterday.toISOString().split("T")[0];

  // 1. Gather metrics from multiple tables in parallel
  const [revenueData, newSubs, publishedContent] = await Promise.all([
    supabase.from("revenue_events")
      .select("amount, source")
      .gte("created_at", dateStr)
      .lt("created_at", new Date().toISOString().split("T")[0]),
    supabase.from("email_subscribers")
      .select("id", { count: "exact" })
      .gte("created_at", dateStr),
    supabase.from("content_pipeline")
      .select("title, channel, engagement")
      .eq("status", "published")
      .gte("published_at", dateStr),
  ]);

  // 2. Calculate summary metrics
  const totalRevenue = revenueData.data?.reduce(
    (sum, e) => sum + e.amount, 0
  ) || 0;
  const newSubscribers = newSubs.count || 0;

  const metrics = {
    revenue: totalRevenue,
    new_subscribers: newSubscribers,
    content_published: publishedContent.data?.length || 0,
    top_content: publishedContent.data?.[0]?.title || "None",
  };

  // 3. Read identity for voice matching
  const identity = await readBrain("identity", "your_identity");

  // 4. Generate briefing with Claude
  const briefing = await askClaude(
    `You are a chief of staff AI. Write a morning briefing email.
Be concise, data-driven, and actionable. Use bullet points.
Voice: ${identity?.voice_style?.tone || "Direct and warm."}`,
    `Yesterday's metrics:
- Revenue: $${totalRevenue.toFixed(2)}
- New subscribers: ${newSubscribers}
- Content published: ${metrics.content_published}

Generate a briefing with: Key Numbers, Notable Events,
Recommended Priorities for Today (max 3).`
  );

  // 5. Store and send
  await supabase.from("daily_briefings").insert({
    briefing_date: dateStr,
    metrics,
    highlights: { content: publishedContent.data },
    priorities: { generated: true },
    full_briefing: briefing,
    sent_at: new Date().toISOString(),
  });

  await sendEmail(
    "you@yourdomain.com",
    `Morning Briefing — ${dateStr}`,
    `<div style="font-family: sans-serif; max-width: 600px;">
      <h2>Morning Briefing</h2>
      ${briefing.replace(/\n/g, "<br>")}
    </div>`
  );

  return new Response(JSON.stringify({ success: true }));
});

Agent 2: Content Writer

Autonomous blog post generator that writes in your voice, optimizes for SEO, and queues posts for publication.

Trigger: Cron (daily 6am) Output: Database + CMS Difficulty: Intermediate Value: 4 hrs/week saved

What It Does

Reads your identity document for voice matching, checks content_performance for trending topics, generates a complete 800-1200 word blog post with title, meta description, and tags, and saves it to your content pipeline as "draft" for review or "ready" for auto-publish.

Database Schema

CREATE TABLE content_pipeline (
  id              uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  title           text NOT NULL,
  slug            text NOT NULL UNIQUE,
  content         text NOT NULL,        -- markdown or HTML body
  meta_description text,                 -- SEO meta
  tags            text[] DEFAULT '{}',

  scheduled_at    timestamptz,
  published_at    timestamptz,
  engagement      jsonb DEFAULT '{}',  -- {views, reads, shares, comments}
  generated_by    text DEFAULT 'content-writer',
  created_at      timestamptz DEFAULT now()
);

CREATE INDEX idx_content_status ON content_pipeline(status);
CREATE INDEX idx_content_channel ON content_pipeline(channel);

CREATE TABLE content_performance (
  id          uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  topic       text NOT NULL,
  avg_views   integer DEFAULT 0,
  avg_reads   integer DEFAULT 0,
  avg_shares  integer DEFAULT 0,
  post_count  integer DEFAULT 0,
  score       numeric(5,2) DEFAULT 0, -- composite engagement score
  updated_at  timestamptz DEFAULT now()
);

Edge Function Code

// supabase/functions/content-writer/index.ts
serve(async (req) => {
  // 1. Read identity + voice from brain
  const identity = await readBrain("identity", "your_identity");

  // 2. Get top-performing topics
  const { data: topTopics } = await supabase
    .from("content_performance")
    .select("topic, score")
    .order("score", { ascending: false })
    .limit(5);

  // 3. Check what we have NOT written about recently
  const { data: recentPosts } = await supabase
    .from("content_pipeline")
    .select("title, tags")
    .eq("channel", "blog")
    .order("created_at", { ascending: false })
    .limit(10);

  // 4. Generate post with Claude
  const result = await askClaude(
    `You are a blog writer for ${identity.core_identity.name}.

VOICE: ${identity.voice_style.tone}
FORBIDDEN PHRASES: ${identity.never_say.join(", ")}
EXPERTISE: ${identity.professional.expertise_areas.join(", ")}
AUDIENCE: ${identity.professional.target_audience}

Write a blog post that is 800-1200 words. Use short paragraphs.
Include a compelling title, meta description (under 160 chars),
and 3-5 tags. Return as JSON:
{title, slug, meta_description, content, tags, word_count}`,

    `Top performing topics: ${topTopics?.map(t => t.topic).join(", ")}
Recent posts (avoid repeating): ${recentPosts?.map(p => p.title).join(", ")}
Write about one of the top topics that hasn't been covered recently.`,
    { maxTokens: 4096, temperature: 0.8 }
  );

  // 5. Parse and store
  const post = JSON.parse(result);
  await supabase.from("content_pipeline").insert({
    ...post,
    channel: "blog",
    status: "draft", // Change to "ready" for auto-publish
    generated_by: "content-writer",
  });

  return new Response(JSON.stringify({ success: true, title: post.title }));
});

Agent 4: Sales Pipeline

Automated lead scoring, segmentation, and follow-up sequences that turn subscribers into customers.

Trigger: Webhook + Cron (daily 7am) Output: Database + Email sequences Difficulty: Advanced Value: $$$$ in recovered revenue

What It Does

Every new subscriber gets an initial lead score. The daily cron recalculates scores based on engagement signals (email opens, link clicks, page visits). When a lead crosses a threshold, it triggers a targeted email sequence. Hot leads (score 80+) get flagged for personal outreach.

Database Schema

CREATE TABLE sequence_enrollments (
  id              uuid DEFAULT gen_random_uuid() PRIMARY KEY,

  sequence_id     uuid REFERENCES nurture_sequences(id),

  status          text DEFAULT 'active', -- 'active', 'completed', 'paused', 'converted'

Agent 5: Health Sentinel

System monitoring and alerting agent that watches your infrastructure 24/7 and wakes you up only when something actually matters.

Trigger: Cron (every 15 min) Output: Database + Alert emails Difficulty: Beginner Value: Peace of mind

What It Does

Every 15 minutes, it pings your critical endpoints (website, edge functions, database), checks response times and status codes, and logs the results. If anything is down or degraded, it sends an immediate alert email. It also tracks uptime percentage over time and generates a weekly reliability report.

Database Schema

CREATE TABLE health_checks (
  id              uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  endpoint_name   text NOT NULL,
  endpoint_url    text NOT NULL,
  status_code     integer,
  response_time_ms integer,
  is_healthy      boolean NOT NULL,
  error_message   text,
  checked_at      timestamptz DEFAULT now()
);

CREATE INDEX idx_health_endpoint ON health_checks(endpoint_name, checked_at DESC);

CREATE TABLE monitored_endpoints (
  id              uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  name            text NOT NULL UNIQUE,
  url             text NOT NULL,
  expected_status integer DEFAULT 200,
  max_response_ms integer DEFAULT 5000,  -- alert if slower than this
  is_critical     boolean DEFAULT true,   -- critical = immediate alert
  is_active       boolean DEFAULT true
);

CREATE TABLE alert_log (
  id              uuid DEFAULT gen_random_uuid() PRIMARY KEY,

  endpoint_name   text NOT NULL,
  message         text NOT NULL,
  notified        boolean DEFAULT false,
  created_at      timestamptz DEFAULT now()

Edge Function Code

// supabase/functions/health-sentinel/index.ts
serve(async (req) => {
  const { data: endpoints } = await supabase
    .from("monitored_endpoints")
    .select("*")
    .eq("is_active", true);

  const results = [];

  for (const endpoint of endpoints.data || []) {
    const start = Date.now();
    let statusCode = 0;
    let isHealthy = false;
    let errorMessage = null;

    try {
      const resp = await fetch(endpoint.url, {
        method: "HEAD",
        signal: AbortSignal.timeout(10000),
      });
      statusCode = resp.status;
      isHealthy = statusCode === endpoint.expected_status;
    } catch (err) {
      errorMessage = err.message;
      isHealthy = false;
    }

    const responseTime = Date.now() - start;
    const isSlow = responseTime > endpoint.max_response_ms;

    // Store health check result
    await supabase.from("health_checks").insert({
      endpoint_name: endpoint.name,
      endpoint_url: endpoint.url,
      status_code: statusCode,
      response_time_ms: responseTime,
      is_healthy: isHealthy && !isSlow,
      error_message: errorMessage,
    });

    // Alert if unhealthy and critical
    if ((!isHealthy || isSlow) && endpoint.is_critical) {
      const alertType = !isHealthy ? "down" : "slow";
      const message = !isHealthy
        ? `${endpoint.name} is DOWN (status: ${statusCode})`
        : `${endpoint.name} is SLOW (${responseTime}ms)`;

      await supabase.from("alert_log").insert({
        alert_type: alertType,
        endpoint_name: endpoint.name,
        message,
      });

      // Send immediate alert email
      await sendEmail(
        "you@yourdomain.com",
        `ALERT: ${message}`,
        `<h2 style="color: red">${message}</h2>
         <p>URL: ${endpoint.url}</p>
         <p>Time: ${new Date().toISOString()}</p>
         <p>Error: ${errorMessage || "HTTP " + statusCode}</p>`
      );
    }

    results.push({ name: endpoint.name, healthy: isHealthy, ms: responseTime });
  }

  return new Response(JSON.stringify({ checks: results }));
});

Deployment Checklist

For each agent, follow this exact sequence:

1 Create the database tables. Run the SQL schema in your Supabase SQL editor.

2 Create the edge function. supabase functions new agent-name in your local project.

3 Write the function code. Copy the template, customize for your use case.

4 Test locally. supabase functions serve agent-name and test with curl.

5 Deploy. supabase functions deploy agent-name --project-ref YOUR_REF

6 Test in production. Call the live URL and verify the database entries.

7 Add the cron schedule. Run the cron.schedule SQL in the SQL editor.

8 Monitor for 48 hours. Check the health_checks and consciousness_stream tables to verify the agent is running correctly on schedule.

Recommended deployment order: Health Sentinel first (so you can monitor everything else), then Morning Briefer (immediate daily value), then Content Writer (builds your audience), then Research Analyst (strategic intelligence), then Sales Pipeline (monetization).
